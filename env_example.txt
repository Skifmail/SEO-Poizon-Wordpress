# -----------------------------------------------------------------------------
# 1. POIZON API (https://poizon-api.com)
# -----------------------------------------------------------------------------
# Получить ключи можно в личном кабинете poizon-api.com
# После регистрации перейдите в раздел "API ключи"

POIZON_API_KEY=ваш_api_ключ_здесь
POIZON_CLIENT_ID=ваш_client_id_здесь

# -----------------------------------------------------------------------------
# 2. WORDPRESS / WOOCOMMERCE API
# -----------------------------------------------------------------------------
# Как получить ключи WooCommerce:
# 1. Войдите в админку WordPress
# 2. WooCommerce → Настройки → Дополнительно → REST API
# 3. Нажмите "Добавить ключ"
# 4. Выберите пользователя и права "Чтение/Запись"
# 5. Скопируйте Consumer Key и Consumer Secret

WC_URL=https://ваш-сайт.ru
WC_CONSUMER_KEY=ck_xxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxx
WC_CONSUMER_SECRET=cs_xxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxx

# -----------------------------------------------------------------------------
# 3. GIGACHAT API (https://developers.sber.ru/gigachat)
# -----------------------------------------------------------------------------
# Как получить ключ GigaChat:
# 1. Зарегистрируйтесь на https://developers.sber.ru
# 2. Создайте проект
# 3. Получите авторизационные данные (Client Secret)
# 4. Вставьте их ниже (одна строка, без переносов!)

GIGACHAT_AUTH_KEY=ваш_очень_длинный_ключ_авторизации_gigachat_здесь

# -----------------------------------------------------------------------------
# 4. НАСТРОЙКИ ПРИЛОЖЕНИЯ
# -----------------------------------------------------------------------------

# Порт для веб-интерфейса (по умолчанию 5000)
FLASK_PORT=5000

# Режим отладки (True для разработки, False для production)
FLASK_DEBUG=True
